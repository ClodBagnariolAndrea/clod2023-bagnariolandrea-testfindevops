const http = require('http'); const hostname = '0.0.0.0'; const port = 3000; const server = http.createServer((req, res) => { res.statusCode = 200; res.setHeader('Content-Type', 'text/plain'); res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Headers', 'Origin, X-Requested-With, Content-Type, Accept'); res.end('Hello World from [YOUR NAME] BackEnd (Version [X])'); }); server.listen(port, hostname, () => { console.log('Server running at http://${hostname}:${port}/'); });
